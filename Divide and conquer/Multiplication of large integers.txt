int get_no_digit(int a)
{
int cnt=0;
while(a!=0)
{

}

}

int get_divisor(int a)
{
const unsigned int B=10;
unsigned int divisor=B;
while(a/divisor>divisor)
	divisor*=B;
return divisor;
}
